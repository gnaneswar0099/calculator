# ------------ Stage 1: Build the project with Maven in Production mode ------------
FROM maven:3.9.4-eclipse-temurin-17 AS build

WORKDIR /app

# Copy and download dependencies
COPY pom.xml . 
RUN mvn dependency:go-offline

# Copy source and build in production mode
COPY . .
RUN mvn clean package -Pproduction -DskipTests

# ------------ Stage 2: Run using Jetty ------------
FROM jetty:11.0.14-jdk17

# Copy WAR from build stage to Jetty's webapps directory
COPY --from=build /app/target/*.war /var/lib/jetty/webapps/ROOT.war

EXPOSE 8080

# Jetty will automatically deploy the WAR
